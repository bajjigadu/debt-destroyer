// ===================== // DEBT DESTROYER v2 CORE // ===================== const bankDatabase = { "CommBank": 18.2, "NAB": 16.5, "Westpac": 15.99, "ANZ": 17.8, "Latitude": 21.5, "Afterpay": 0, "Zip Pay": 0, "HSBC": 14.9, "Coles Credit": 19.99, "Kogan Card": 22.9, "Flexi Loan": 17.5, "Other Credit Card": 18.0, "Personal Loan": 12.5 }; const bankList = Object.keys(bankDatabase); const money = n => '$' + Math.round(n || 0).toLocaleString(); // ===================== // STATE // ===================== let state = { passcode: localStorage.getItem('appPasscode') || '1234', bank: parseFloat(localStorage.getItem('bankBalance') || 0), saved: parseFloat(localStorage.getItem('totalSaved') || 0), goal: parseFloat(localStorage.getItem('savingsGoal') || 50), start: new Date(localStorage.getItem('planStartDate') || '2026-05-27'), term: parseInt(localStorage.getItem('termYears') || 5), debts: JSON.parse(localStorage.getItem('debtsData') || '[]'), originalDebt: parseFloat(localStorage.getItem('originalDebt') || 0), history: JSON.parse(localStorage.getItem('history') || '[]'), light: localStorage.getItem('isLightMode') === 'true' }; // ===================== // TOAST // ===================== function toast(msg){ const el = document.getElementById('toast'); el.innerText = msg; el.style.display = 'block'; clearTimeout(window.__t); window.__t = setTimeout(()=> el.style.display='none', 1800); } // ===================== // SAVE // ===================== function save(){ localStorage.setItem('bankBalance', state.bank); localStorage.setItem('totalSaved', state.saved); localStorage.setItem('savingsGoal', state.goal); localStorage.setItem('planStartDate', state.start.toISOString().slice(0,10)); localStorage.setItem('termYears', state.term); localStorage.setItem('debtsData', JSON.stringify(state.debts)); localStorage.setItem('originalDebt', state.originalDebt); localStorage.setItem('history', JSON.stringify(state.history)); localStorage.setItem('appPasscode', state.passcode); localStorage.setItem('isLightMode', state.light); } // ===================== // CALCS // ===================== function totalDebt(){ return state.debts.reduce((s,d)=>s+(+d.balance||0),0); } function autoWeekly(){ const t = totalDebt(); if(!t) return 0; return Math.ceil(t / (state.term * 52)); } function distributeMin(){ const t = totalDebt(); const w = autoWeekly(); state.debts.forEach(d=>{ d.min = t ? Math.max(5, Math.round((d.balance/t)*w)) : 0; }); } // ===================== // RENDER // ===================== function renderDebts(){ const tbody = document.getElementById('debtTableBody'); tbody.innerHTML = ''; if(!state.debts.length){ tbody.innerHTML = 'Add a debt to start'; return; } state.debts.forEach((d,i)=>{ const tr = document.createElement('tr'); tr.innerHTML = ` ${money(d.min||0)} `; const bank = tr.querySelector('.bank-select'); const bal = tr.querySelector('.balance'); const rate = tr.querySelector('.rate'); const pay = tr.querySelector('.pay'); const del = tr.querySelector('.del'); const update = ()=>{ d.bank = bank.value; d.balance = Math.max(0, parseFloat(bal.value)||0); d.rate = parseFloat(rate.value)||0; distributeMin(); sync(); }; bank.onchange = ()=>{ rate.value = bankDatabase[bank.value]; update(); }; bal.oninput = debounce(update, 200); rate.oninput = debounce(update, 200); pay.onclick = ()=>{ if(d.balance<=0) return; const amt = Math.min(50,d.balance); d.balance -= amt; state.bank -= amt; state.history.push({type:'pay',amt,debt:d.bank,date:Date.now()}); toast(`Paid ${money(amt)} → ${d.bank}`); sync(); }; del.onclick = ()=>{ state.debts.splice(i,1); sync(); }; tbody.appendChild(tr); }); distributeMin(); syncStats(); } // ===================== // SYNC UI // ===================== function syncStats(){ const t = totalDebt(); document.getElementById('heroTotalDebt').innerText = money(t); document.getElementById('heroWeeklyPayment').innerText = money(autoWeekly()); document.getElementById('heroBankDisplay').innerText = money(state.bank); document.getElementById('totalSavingsDisplay').innerText = money(state.saved); document.getElementById('debtCountBadge').innerText = state.debts.length; } // full sync function sync(){ renderDebts(); syncStats(); save(); } // ===================== // DEBOUNCE // ===================== function debounce(fn,ms){ let t; return (...args)=>{ clearTimeout(t); t=setTimeout(()=>fn(...args),ms); }; } // ===================== // ACTIONS // ===================== function addDebt(){ state.debts.push({ bank:'Other Credit Card', balance:0, rate:18, min:0 }); sync(); } function quickPay(){ const w = autoWeekly(); if(!w) return toast("No active debt"); let pay = Math.min(state.bank,w); if(pay<=0) return toast("Insufficient balance"); state.bank -= pay; state.saved += pay; let remaining = pay; state.debts.sort((a,b)=>b.rate-a.rate).forEach(d=>{ if(!remaining) return; let p = Math.min(d.balance,remaining); d.balance -= p; remaining -= p; }); toast(`Auto paid ${money(pay)}`); sync(); } // ===================== // INIT // ===================== function init(){ document.body.classList.toggle('light',state.light); document.getElementById('addDebtRowBtn').onclick = addDebt; document.getElementById('quickAddPaymentBtn').onclick = quickPay; document.getElementById('manualSalaryBtn').onclick = ()=>{ state.bank += state.goal; state.saved += state.goal; toast("Payday added"); sync(); }; document.getElementById('resetFullBtn').onclick = ()=>{ if(confirm("Reset everything?")){ localStorage.clear(); location.reload(); } }; sync(); } init();
